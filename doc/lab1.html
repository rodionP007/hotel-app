554    def validate_client_input(self):
555        fio = self.lineFIO_clientEdit.text().strip()
556        if not fio:
557            raise ErrorFieldEmpty("ФИО клиента")
558
559        parts = fio.split()
560        if len(parts) < 2:
561            raise ErrorInvalidFormat("ФИО клиента", "Введите хотя бы фамилию и имя")
562
563        for part in parts:
564            if not re.match(r"^[А-Яа-яЁёA-Za-z\-]+$", part):
565                raise ErrorInvalidFormat("ФИО клиента", "Некорректные символы, используйте только буквы")
566            if len(part) < 2:
567                raise ErrorInvalidFormat("ФИО клиента", f"Слишком короткое слово '{part}'")
568
569        date_start = self.dateClientEdit.date()
570        if not date_start.isValid():
571            raise ErrorDateField("Дата заезда", "Поле не может быть пустым")
572
573        today = QtCore.QDate.currentDate()
574        max_date = today.addYears(2)
575        if date_start > max_date:
576            raise ErrorDateField("Дата заезда", "Слишком далёкая дата")
577        if date_start < today.addYears(-1):
578            raise ErrorDateField("Дата заезда", "Слишком старая дата")
579
580        return fio, date_start

616    def _load_csv_into_clients_table(self, filepath, delimiter=";"):
617        """
618        Читает CSV и заполняет tableWidgetClients.
619        Ожидаемые колонки (в заголовке): ФИО, Номер, Дата заезда, Дата выезда
620        Если заголовки отличаются, попытаемся найти похожие по имени.
621        """
622        try:
623            with open(filepath, newline='', encoding='utf-8') as f:
624                reader = csv.reader(f, delimiter=';')
625                rows = list(reader)
626                if not rows:
627                    QtWidgets.QMessageBox.warning(self.window, "Файл пуст", "Выбранный CSV файл пуст.")
628                    return
629
630                header = [h.strip() for h in rows[0]]
631                data = rows[1:]
632
633
634
635                # Простейшая стратегия: ищем индексы по содержанию заголовка
636                def find_col(possible_names):
637                    for i, h in enumerate(header):
638                        low = h.lower()
639                        for p in possible_names:
640                            if p in low:
641                                return i
642                    return None
643
644                # наилучшие кандидаты:
645                idx_fio = find_col(["фио", "фамилия", "fio", "name"])
646                idx_room = find_col(["номер", "room", "room_no"])
647                idx_start = find_col(["дата заезда", "checkin", "date_start", "date"])
648                idx_end = find_col(["дата выезда", "checkout", "date_end"])
649
650                # Если вообще не нашли, просто используем первые 4 колонки
651                if idx_fio is None and len(header) >= 1:
652                    idx_fio = 0
653                if idx_room is None and len(header) >= 2:
654                    idx_room = 1
655                if idx_start is None and len(header) >= 3:
656                    idx_start = 2
657                if idx_end is None and len(header) >= 4:
658                    idx_end = 3
659
660                # Очищаем таблицу
661                self.tableWidgetClients.setRowCount(0)
662
663                for row in data:
664                    # безопасно получить значение по индексу
665                    def get_val(i):
666                        return row[i].strip() if i is not None and i < len(row) else ""
667
668                    fio = get_val(idx_fio)
669                    room = get_val(idx_room)
670                    start = get_val(idx_start)
671                    end = get_val(idx_end)
672
673                    self.add_client_row(fio, room,
674                                        QtCore.QDate.fromString(start, "dd.MM.yyyy") if start else QtCore.QDate(),
675                                        QtCore.QDate.fromString(end, "dd.MM.yyyy") if end else QtCore.QDate())
676
677                QtWidgets.QMessageBox.information(self.window, "Успешно",
678                                                  f"Загружено {len(data)} строк в таблицу клиентов.")
679        except Exception as e:
680            QtWidgets.QMessageBox.critical(self.window, "Ошибка при чтении CSV", str(e))

622        """

625                reader = csv.reader(f, delimiter=';')

627                if not rows:
628                    QtWidgets.QMessageBox.warning(self.window, "Файл пуст", "Выбранный CSV файл пуст.")
629                    return
630
631                header = [h.strip() for h in rows[0]]
632                data = rows[1:]
633
634
635
636                # Простейшая стратегия: ищем индексы по содержанию заголовка
637                def find_col(possible_names):
638                    for i, h in enumerate(header):
639                        low = h.lower()
640                        for p in possible_names:
641                            if p in low:
642                                return i
643                    return None
644
645                # наилучшие кандидаты:
646                idx_fio = find_col(["фио", "фамилия", "fio", "name"])
647                idx_room = find_col(["номер", "room", "room_no"])

649                idx_end = find_col(["дата выезда", "checkout", "date_end"])

652                if idx_fio is None and len(header) >= 1:
653                    idx_fio = 0
654                if idx_room is None and len(header) >= 2:
655                    idx_room = 1
656                if idx_start is None and len(header) >= 3:
657                    idx_start = 2
658                if idx_end is None and len(header) >= 4:
659                    idx_end = 3
660
661                # Очищаем таблицу
662                self.tableWidgetClients.setRowCount(0)
663
664                for row in data:
665                    # безопасно получить значение по индексу
666                    def get_val(i):
667                        return row[i].strip() if i is not None and i < len(row) else ""
668
669                    fio = get_val(idx_fio)
670                    room = get_val(idx_room)
671                    start = get_val(idx_start)
672                    end = get_val(idx_end)
673
674                    self.add_client_row(fio, room,
675                                        QtCore.QDate.fromString(start, "dd.MM.yyyy") if start else QtCore.QDate(),
676                                        QtCore.QDate.fromString(end, "dd.MM.yyyy") if end else QtCore.QDate())

678                QtWidgets.QMessageBox.information(self.window, "Успешно",
679                                                  f"Загружено {len(data)} строк в таблицу клиентов.")

555    def validate_client_input(self):
556        fio = self.lineFIO_clientEdit.text().strip()
557        if not fio:
558            raise ErrorFieldEmpty("ФИО клиента")
559
560        parts = fio.split()
561        if len(parts) < 2:
562            raise ErrorInvalidFormat("ФИО клиента", "Введите хотя бы фамилию и имя")
563
564        for part in parts:
565            if not re.match(r"^[А-Яа-яЁёA-Za-z\-]+$", part):
566                raise ErrorInvalidFormat("ФИО клиента", "Некорректные символы, используйте только буквы")
567            if len(part) < 2:
568                raise ErrorInvalidFormat("ФИО клиента", f"Слишком короткое слово '{part}'")
569
570        date_start = self.dateClientEdit.date()
571        if not date_start.isValid():
572            raise ErrorDateField("Дата заезда", "Поле не может быть пустым")
573
574        today = QtCore.QDate.currentDate()
575        max_date = today.addYears(2)
576        if date_start > max_date:
577            raise ErrorDateField("Дата заезда", "Слишком далёкая дата")
578        if date_start < today.addYears(-1):
579            raise ErrorDateField("Дата заезда", "Слишком старая дата")
580
581        return fio, date_start
